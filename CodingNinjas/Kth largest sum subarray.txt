1.  K-th Largest Sum Subarray

#include<queue>
#include<algorithm>
int getKthLargest(vector<int> &arr, int k)
{
	
	priority_queue<int, vector<int> , greater<int>> sumStore;
	
	for(int i = 0 ; i< arr.size(); i++)
	{
		int sum = 0;
		for(int j = i ; j< arr.size(); j++)
		{
			sum += arr[j];
			if(sumStore.size() < k)
			{
				sumStore.push(sum);

			}
			else
			{
				if(sum > sumStore.top())
				{
					sumStore.pop();
					sumStore.push(sum);
				}
			}
		}
		
	}

	

	return sumStore.top();

}
